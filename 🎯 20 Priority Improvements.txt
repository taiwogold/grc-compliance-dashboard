6. Database Connection Pooling
Current Problem: Opens/closes SQLite connection on every function call Solution:

import contextlib

@contextlib.contextmanager
def get_db_connection(db_path: Path = DB_PATH):
    conn = sqlite3.connect(str(db_path), check_same_thread=False)
    try:
        yield conn
    finally:
        conn.close()

# Usage:
with get_db_connection() as conn:
    df = pd.read_sql_query("SELECT * FROM snapshots", conn)

def validate_uploaded_csv(uploaded_file):
    """Validate uploaded CSV for security & format."""
    # Check file size (max 10MB)
    if uploaded_file.size > 10 * 1024 * 1024:
        raise ValueError("File too large (max 10MB)")

    # Check file extension
    if not uploaded_file.name.endswith('.csv'):
        raise ValueError("Only CSV files allowed")

    # Parse and validate structure
    df = pd.read_csv(uploaded_file)

    # Check for required columns

    if missing:

10. Logging - Inconsistent Implementation
Current Problem: Some modules use logger, some use print, some have no logging Solution:

python

# utils/logger.py
import logging
from pathlib import Path

def setup_logger(name: str, log_file: str = "app.log") -> logging.Logger:
    logger = logging.getLogger(name)
    logger.setLevel(logging.INFO)

    # File handler
    fh = logging.FileHandler(Path("logs") / log_file)
    fh.setFormatter(logging.Formatter(
        '%(asctime)s - %(name)s - %(levelname)s - %(message)s'
    ))
    logger.addHandler(fh)

    return logger

# Use everywhere:
logger = setup_logger(__name__)
